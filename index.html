async function cadastrarUsuario(data) { const { error } = await window.supabaseClient .from("users") .insert([{ name: data.name, email: data.email, company: data.company || null, whatsapp: data.whatsapp || null, role: "client", status: "pending", plan: "starter" }]); if (error) { console.error(error); alert("Erro ao cadastrar usuário."); return false; } alert("Cadastro enviado para aprovação!"); return true; }async function loginUsuario(email) { const { data, error } = await window.supabaseClient .from("users") .select("*") .eq("email", email) .single(); if (error || !data) { alert("Usuário não encontrado."); return null; } if (data.status !== "active" && data.role !== "admin") { alert("Conta aguardando aprovação do administrador."); return null; } return data; }async function carregarClientes() { const { data, error } = await window.supabaseClient .from("users") .select("*") .order("created_at", { ascending: false }); if (error) { console.error(error); return []; } return data; }
